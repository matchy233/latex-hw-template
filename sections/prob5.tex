\begin{homeworkProblem}

\solution
\begin{proof}
We know that  $B = Q^{-1}AQ$. Then we have
\[
\begin{aligned}
    tr(B) &= tr((Q^{-1})(AQ)) \\
          &= tr((AQ)(Q^{-1})) & \text{(By problem (1))}\\
          &= tr(A(QQ^{-1})) \\
          &= tr(A)
\end{aligned}
\]
\end{proof}

\end{homeworkProblem}
