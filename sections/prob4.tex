\begin{homeworkProblem}

Given \[
\begin{aligned}
    &\mathbf{V} = \mathbb{R}^3 \\
    &S = \left\{ (1,0,1), (0,1,1), (1,3,3)\right\} \\
    &x = (1,1,2)
\end{aligned},
\]apply the Gram-Schmidt process to the subset $S$ of the inner product space $\mathbf{V}$ to obtain an orthogonal basis for $\text{span}(S)$. Then normalize the vectors in this basis to obtain an orthonormal basis $\beta$ for $\text{span}(S)$, and compute the Fourier coefficients of the given vector $x$ relative to $\beta$. Finally, use the following theorem to verify your results:

Let $\mathbf{V}$ be a nonzero finite-dimensional inner product space. Then $\mathbf{V}$ has an orthonormal basis $\beta$. Furthermore, if $\beta = \{v_1, v_2, \cdots, v_n\}$ and $x \in \mathbf{V}$, then
\[
    x = \sum_{i=1}^n \langle x, v_i \rangle v_i .
\]

\solution

\part: Apply the Gram-Schmidt process to obtain the orthonormal basis $\beta$

Suppose $S' = \{v_1, v_2, v_3\}$ and $S = \{w_1, w_2, w_3\}$. We have $v_1 = w_1 = (1, 0, 1)$. And \[
\begin{aligned}
v_2 &= w_2 - \frac{\innerprod[w_2]{v_1}}{\norm{v_1}^2} v_1 \\
&= (0, 1, 1) - \frac{\innerprod[(0, 1, 1)]{(1, 0, 1)}}{\innerprod[(1,0,1)]{(1,0,1)}} (1,0,1) \\
&= (0, 1, 1) - \frac{1}{2}(1, 0, 1) \\
&= \left(-\frac{1}{2}, 1, \frac{1}{2}\right)
\end{aligned}
\]

\[
\begin{aligned}
v_3 &= w_3 - \frac{\innerprod[v_3]{v_1}}{\norm{v_1}^2}v_1 - \frac{\innerprod[v_3]{v_2}}{\norm{v_2}}v_2 \\
&= (1, 3, 3) - \frac{\innerprod[(1,3,3)]{(1,0,1)}}{\innerprod[(1,0,1)]{(1,0,1)}} - \frac{\innerprod[(1,3,3)]{\left(-\frac{1}{2}, 1, \frac{1}{2}\right)}}{\innerprod[\left(-\frac{1}{2}, 1, \frac{1}{2}\right)]{\left(-\frac{1}{2}, 1, \frac{1}{2}\right)}} \\
&= \left(\frac{1}{3}, \frac{1}{3}, -\frac{1}{3}\right)
\end{aligned}
\]

These vectors can be normalized to obtain the orthonormal basis $\beta = \{u_1, u_2, u_3\}$, where

\[
\begin{aligned}
    u_1 &= \frac{1}{\norm{v_1}} v_1 = \frac{1}{\sqrt{2}}\left(1, 0, 1\right) \\
    u_2 &= \frac{1}{\norm{v_2}} v_2 = \frac{\sqrt{6}}{3} \left(-\frac{1}{2}, 1, \frac{1}{2}\right)\\
    u_3 &= \frac{1}{\norm{v_3}} v_3 = \sqrt{3} \left(\frac{1}{3}, \frac{1}{3}, -\frac{1}{3}\right)
\end{aligned}
\]

\part: Compute the Fourier coefficients w.r.t $x$

\[
\begin{aligned}
c_1 &= \innerprod[x]{u_1} = \frac{3\sqrt{2}}{2}\\
c_2 &= \innerprod[x]{u_2} = \frac{\sqrt{6}}{2}\\
c_3 &= \innerprod[x]{u_3} = 0
\end{aligned}
\]

\part: Verify the results

\[
\begin{aligned}
\innerprod[x]{u_1}u_1 + \innerprod[x]{u_2}u_2 + \innerprod[x]{u_3}u_3
&= \frac{3}{2}\left(1, 0, 1\right)
   + \left(-\frac{1}{2}, 0, \frac{1}{2}\right)
   + (0, 0, 0)\\
&= (1, 1, 2) = x
\end{aligned}
\]

\end{homeworkProblem}
